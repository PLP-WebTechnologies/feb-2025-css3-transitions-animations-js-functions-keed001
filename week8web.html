// Function to store user preferences in localStorage function saveUserPreference(prefKey, prefValue) { localStorage.setItem(prefKey, prefValue); } // Function to retrieve user preferences from localStorage function getUserPreference(prefKey) { return localStorage.getItem(prefKey); } // Function to apply animation based on user action function triggerAnimation(elementId) { const element = document.getElementById(elementId); if (element) { element.classList.add('animated'); } } // Example Usage document.getElementById('animateButton').addEventListener('click', function() { saveUserPreference('animationTriggered', 'true'); // Store user action triggerAnimation('targetElement'); // Apply animation }); // Check stored preference on page load window.onload = function() { if (getUserPreference('animationTriggered') === 'true') { triggerAnimation('targetElement'); } };
